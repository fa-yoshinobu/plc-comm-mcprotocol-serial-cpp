80[[nodiscard]] inline bool parse_u32(
81 std::string_view text,
82 std::uint32_t& out_value,
83 int base) noexcept {
84 if (text.empty()) {
85 return false;
86 }
87
88 std::uint32_t value = 0U;
89 for (char ch : text) {
90 std::uint32_t digit = 0U;
91 if (ch >= '0' && ch <= '9') {
92 digit = static_cast<std::uint32_t>(ch - '0');
93 } else if (base == 16 && ch >= 'A' && ch <= 'F') {
94 digit = static_cast<std::uint32_t>(ch - 'A' + 10);
95 } else if (base == 16 && ch >= 'a' && ch <= 'f') {
96 digit = static_cast<std::uint32_t>(ch - 'a' + 10);
97 } else {
98 return false;
99 }
100
101 if (digit >= static_cast<std::uint32_t>(base)) {
102 return false;
103 }
104
105 value = static_cast<std::uint32_t>(value * static_cast<std::uint32_t>(base) + digit);
106 }
107
108 out_value = value;
109 return true;
110}

200[[nodiscard]] inline Status parse_device_address(
201 std::string_view text,
202 DeviceAddress& out_device) noexcept {
203 for (const auto& spec : detail::kDeviceParseSpecs) {
204 if (text.size() <= spec.prefix_length) {
205 continue;
206 }
207
208 bool prefix_match = true;
209 for (std::size_t index = 0; index < spec.prefix_length; ++index) {
210 const char lhs = detail::ascii_upper(text[index]);
211 if (lhs != spec.prefix[index]) {
212 prefix_match = false;
213 break;
214 }
215 }
216 if (!prefix_match) {
217 continue;
218 }
219
220 std::uint32_t number = 0;
221 if (!detail::parse_u32(text.substr(spec.prefix_length), number, spec.base)) {
222 return make_status(StatusCode::InvalidArgument, "Device address number is invalid");
223 }
224
225 out_device = DeviceAddress {
226 .code = spec.code,
227 .number = number,
228 };
229 return ok_status();
230 }
231
232 return make_status(StatusCode::InvalidArgument, "Device address prefix is not supported");
233}

361 std::span<const RandomReadSpec> specs,
362 std::span<RandomReadItem> out_items,
363 RandomReadRequest& out_request) noexcept {
364 if (out_items.size() < specs.size()) {
365 return make_status(StatusCode::BufferTooSmall, "Random read output item buffer is too small");
