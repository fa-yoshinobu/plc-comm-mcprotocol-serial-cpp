80[[nodiscard]] inline bool parse_u32(
81 std::string_view text,
82 std::uint32_t& out_value,
83 int base) noexcept {
84 if (text.empty()) {
85 return false;
86 }
87
88 std::uint32_t value = 0U;
89 for (char ch : text) {
90 std::uint32_t digit = 0U;
91 if (ch >= '0' && ch <= '9') {
92 digit = static_cast<std::uint32_t>(ch - '0');
93 } else if (base == 16 && ch >= 'A' && ch <= 'F') {
94 digit = static_cast<std::uint32_t>(ch - 'A' + 10);
95 } else if (base == 16 && ch >= 'a' && ch <= 'f') {
96 digit = static_cast<std::uint32_t>(ch - 'a' + 10);
97 } else {
98 return false;
99 }
100
101 if (digit >= static_cast<std::uint32_t>(base)) {
102 return false;
103 }
104
105 value = static_cast<std::uint32_t>(value * static_cast<std::uint32_t>(base) + digit);
106 }
107
108 out_value = value;
109 return true;
110}

190[[nodiscard]] inline Status parse_device_address(
191 std::string_view text,
192 DeviceAddress& out_device) noexcept {
193 for (const auto& spec : detail::kDeviceParseSpecs) {
194 if (text.size() <= spec.prefix_length) {
195 continue;
196 }
197
198 bool prefix_match = true;
199 for (std::size_t index = 0; index < spec.prefix_length; ++index) {
200 const char lhs = detail::ascii_upper(text[index]);
201 if (lhs != spec.prefix[index]) {
202 prefix_match = false;
203 break;
204 }
205 }
206 if (!prefix_match) {
207 continue;
208 }
209
210 std::uint32_t number = 0;
211 if (!detail::parse_u32(text.substr(spec.prefix_length), number, spec.base)) {
212 return make_status(StatusCode::InvalidArgument, "Device address number is invalid");
213 }
214
215 out_device = DeviceAddress {
216 .code = spec.code,
217 .number = number,
218 };
219 return ok_status();
220 }
221
222 return make_status(StatusCode::InvalidArgument, "Device address prefix is not supported");
223}

351 if (out_items.size() < specs.size()) {
